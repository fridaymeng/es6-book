Avoid IIFEs in ES6

In ES5, you had to use an IIFE if you wanted to keep a variable local:

(function () {  // open IIFE
    var tmp = ···;
    ···
}());  // close IIFE
console.log(tmp); // ReferenceError

In ECMAScript 6, you can simply use a block and a let or const declaration:

{  // open block
    let tmp = ···;
    ···
}  // close block
console.log(tmp); // ReferenceError

// my_module.js

// Module-private variable:
let countInvocations = 0;

export function myFunc(x) {
    countInvocations++;
    ···
}

import { myFunc } from 'my_module.js'

myFunc(33)

Function expression:

Generator functions

Generator function expression:

  const foo = function* (x) { ··· };

Generator function declaration:

  function* foo(x) { ··· }

Method definitions

const obj = {
  add(x, y) {
    return x + y
  }, // comma is required
  sub(x, y) {
    return x - y
  } // comma is optional
}

class AddSub {
  add(x, y) {
    return x + y
  } // no comma
  sub(x, y) {
    return x - y
  } // no comma
}

Generator method definitions

const obj = {

        ···
    },
};

        ···
    }
}

Classes

// Base class: no `extends`
class Point {
  constructor(x, y) {
    this.x = x
    this.y = y
  }
  toString() {
    return `(${this.x}, ${this.y})`
  }
}

// This class is derived from `Point`
class ColorPoint extends Point {
  constructor(x, y, color) {
    super(x, y)
    this.color = color
  }
  toString() {
    return super.toString() + ' in ' + this.color
  }
}
